def check_lib_used(lib_list, lib_name):
    for root, dirs, files in os.walk(target_path, followlinks=False):
        if lib_name in files:
            lib_linked = ''
            file_path = os.path.join(root, lib_name)
            result_linked = os.popen('file %s' % file_path).readlines()
            for result_line in result_linked:
                if 'symbolic link to' in result_line:
                    m_linked = re.match('^.* symbolic link to (?P<lib>.*).*', result_line)
                    if(m_linked):
                        lib_linked = m_linked.group('lib')
                        if lib_linked!='':
                            if lib_linked not in lib_list:
                                lib_list.append(lib_linked)
                            lib_depend.write('>>> %s -> %s\n' % (lib_name, lib_linked))

            result = os.popen('readelf -d %s' %file_path).readlines()
            for line in result:
                if 'Shared library:' in line:
                    m = re.match('^.*Shared library: \[(?P<lib>.*)\].*', line)
                    if(m):
                        found = m.group('lib')
                        if found not in lib_list:
                            lib_list.append(found)
                            check_lib_used(lib_list, found)
                        lib_depend.write('%s -> %s\n' % (lib_name, found))

for root, dirs, files in os.walk(target_path, followlinks=False):

def analyze_elf(elf_file):
    top_counts = 20
    kernel_lists = []
    section_lists = ".head.text .init.text .init.data .text .rodata __ksymtab __ksymtab_gpl __ksymtab_strings __param __modver .data .notes __ex_table .bss .u_boot_list .csky_stack_size"
    elf_header_fmt = ' *\[(?P<nr>[ 0-9]*)\] *(?P<name>[.\_a-z]*) *(?P<type>[\_A-Z]*) *(?P<addr>[0-9a-f]*) *(?P<off>[0-9a-f]*) *(?P<size>[0-9a-f]*).*'
    elf_symbol_fmt = ' *(?P<num>[0-9]*): (?P<value>[0-9abcdef]*) *(?P<size>[0-9]*).*'
    section_total_size = 0
    elf_output_name = os.path.split(elf_file)[1]

    section_lines = os.popen('csky-abiv2-linux-readelf -S %s' % elf_file).readlines()
    for section_line in section_lines:
        m = re.match(elf_header_fmt, section_line)
        if m:
            section_name = m.group('name')
            if section_name in section_lists and section_name != '':
                section_size = int(m.group('size'), 16)
                section_total_size += section_size
                print ('%20s: %8d KB' % (section_name, section_size/1024))
    print ('%20s: %8d KB' % ('Total', section_total_size/1024))

    kernel_lines = os.popen('csky-abiv2-linux-readelf -s %s' % elf_file).readlines()
    for kernel_line in kernel_lines:
        m = re.match(elf_symbol_fmt, kernel_line)
        if m:
            kernel_line_list = re.split(r'\s+', kernel_line)
            if kernel_line_list[3][0:2] == '0x':
                kernel_line_list[3] = int(kernel_line_list[3][2:], 16)
            kernel_lists.append([kernel_line_list[8], kernel_line_list[3], kernel_line_list[4], kernel_line_list[2]])

    kernel_data = pd.DataFrame(np.asarray(kernel_lists), columns=['name', 'size', 'type', 'address'])
    kernel_data['size'] = kernel_data['size'].astype(int)

    kernel_all_data = pd.DataFrame(kernel_data.sort_values(by=['size', 'type'], ascending=[False, False]))
    kernel_func_data = kernel_data[kernel_data['type'] == 'FUNC'].sort_values(by=['size'], ascending=False)
    kernel_object_data = kernel_data[kernel_data['type'] == 'OBJECT'].sort_values(by=['size'], ascending=False)

    elf_types = kernel_data['type'].unique()
    totalsize_of_types = 0

    print '\nSize of %s:' % (elf_types)
    for type in elf_types:
        size_of_type = np.asarray(kernel_data[kernel_data['type'] == type].sort_values(by=['size'], ascending=False)['size']).sum()
        print ('%12s: %6.2f KB' % (type, size_of_type/1024))
        totalsize_of_types += size_of_type
    print ('%12s: %6.2f KB' % ('Total', totalsize_of_types/1024))

    kernel_all_data.to_csv(('%s/%s_all.csv' % (output_dir, elf_output_name)), index=0)
    kernel_all_data.head(top_counts)

    kernel_func_data.to_csv(('%s/%s_func.csv' % (output_dir, elf_output_name)), index=0)
    kernel_func_data.head(top_counts)

    kernel_object_data.to_csv(('%s/%s_data.csv' % (output_dir, elf_output_name)), index=0)
    kernel_object_data.head(top_counts)
